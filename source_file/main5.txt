let number = 7;

for i = 0 to 10
{
    if(i / 2 == 0)
    {
        print(string(i) + " is even, skipping.");
        continue;
    }

    if(i == number)
    {
        print("Found the target number " + string(i));
        break;
    }
    else
    {
        print(string(i) + " is not the target number.");
    }
}
